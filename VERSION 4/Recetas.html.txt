function buscarSkuProducto(q){
  var dd = document.getElementById('sku-dropdown');
  if(!q || q.length<2){ dd.classList.remove('open'); return; }
  var ql = q.toLowerCase();
  var matches = productos.filter(function(p){
    return p.sku.toLowerCase().indexOf(ql)!==-1 ||
           p.modelo.toLowerCase().indexOf(ql)!==-1 ||
           p.color.toLowerCase().indexOf(ql)!==-1 ||
           p.categoria.toLowerCase().indexOf(ql)!==-1;
  }).slice(0,12);

  if(!matches.length){ dd.classList.remove('open'); return; }

  dd.innerHTML = matches.map(function(p){
    var esOk = p.disponible > 3;
    var esLow = p.disponible > 0 && p.disponible <= 3;
    var sColor = esOk?'var(--green)':esLow?'var(--warn)':'var(--red)';
    return '<div class="sku-item" onclick="seleccionarSku(\''+esc(p.sku)+'\')">'+
      '<div>'+
        '<div class="si-sku">'+esc(p.sku)+'</div>'+
        '<span class="cdot" style="background:'+colorHex(p.color)+';margin-top:4px;display:block"></span>'+
      '</div>'+
      '<div>'+
        '<div class="si-name">'+esc(p.modelo)+'</div>'+
        '<div class="si-det">'+cdotHtml(p.color)+esc(p.color)+' · '+esc(p.talla)+' · '+esc(p.categoria)+'</div>'+
        (p.precioDetal>0?'<div class="si-det" style="color:var(--red);margin-top:2px">Detal: '+fmt(p.precioDetal)+(p.precioCosto>0?' · Costo actual: '+fmt(p.precioCosto):'')+'</div>':'')+
      '</div>'+
      '<div class="si-stock" style="color:'+sColor+'">'+p.disponible+' uds</div>'+
    '</div>';
  }).join('');
  dd.classList.add('open');
}

//  CÁLCULO DE COSTO PREVIEW
// ═══════════════════════════════════════════════
function calcularCostoPreview(){
  var metros = parseFloat(document.getElementById('f-metros').value)||0;
  var precioM = parseFloat(document.getElementById('f-precio-metro').value)||0;
  var otros = parseFloat(document.getElementById('f-otros-costos').value)||0;
  var pd = parseFloat(document.getElementById('f-precio-detal').value)||0;
  var pcActual = parseFloat(document.getElementById('f-precio-costo-actual').value)||0;

  if(metros<=0 && precioM<=0 && otros<=0){
    document.getElementById('costo-preview').classList.remove('visible');
    return;
  }
  document.getElementById('costo-preview').classList.add('visible');

  var costoTela = metros * precioM;
  var costoTotal = costoTela + otros;

  document.getElementById('cp-tela').textContent = fmt(costoTela);
  document.getElementById('cp-otros').textContent = fmt(otros);
  document.getElementById('cp-total').textContent = fmt(costoTotal);

  // Margen
  var margenTxt = '';
  if(costoTotal>0 && pd>0){
    var gan = pd - costoTotal;
    var pct = Math.round((gan/pd)*100);
    var col = gan>=0?'var(--green)':'var(--red)';
    margenTxt = 'Vs precio detal '+fmt(pd)+': <strong style="color:'+col+'">'+
      (gan>=0?'+':'')+fmt(gan)+' ('+pct+'%)</strong>';
  }
  document.getElementById('cp-margen-txt').innerHTML = margenTxt;

  // Botón actualizar si hay SKU seleccionado y costo calculable
  var updateWrap = document.getElementById('cp-update-wrap');
  if(skuSeleccionado && costoTotal>0){
    updateWrap.style.display='block';
    var lbl = '';
    if(pcActual>0){
      var diff = costoTotal - pcActual;
      lbl = 'Costo actual: '+fmt(pcActual)+(diff!==0?' → diferencia: '+(diff>0?'+':'')+fmt(diff):'');
    } else {
      lbl = 'El producto no tiene costo registrado';
    }
    document.getElementById('cp-update-lbl').textContent = lbl;
  } else {
    updateWrap.style.display='none';
  }
}

//  ACTUALIZAR COSTO EN EL PRODUCTO
// ═══════════════════════════════════════════════
function actualizarCostoProducto(){
  if(!skuSeleccionado) return;
  var metros = parseFloat(document.getElementById('f-metros').value)||0;
  var precioM = parseFloat(document.getElementById('f-precio-metro').value)||0;
  var otros = parseFloat(document.getElementById('f-otros-costos').value)||0;
  var costoTotal = (metros*precioM)+otros;
  if(costoTotal<=0){ toast('El costo calculado es 0','err'); return; }

  var sku = skuSeleccionado.sku;
  var p = skuSeleccionado;

  showLoader('Actualizando precio de costo...');
  google.script.run
    .withSuccessHandler(function(res){
      hideLoader();
      if(res.ok){
        // Actualizar local
        document.getElementById('f-precio-costo-actual').value = costoTotal;
        skuSeleccionado.precioCosto = costoTotal;
        var pp = productos.find(function(x){return x.sku===sku;});
        if(pp) pp.precioCosto = costoTotal;
        toast('✓ Costo de '+sku+' actualizado a '+fmt(costoTotal),'ok');
        document.getElementById('cp-update-lbl').textContent = 'Costo actualizado: '+fmt(costoTotal);
        calcularCostoPreview();
      } else { toast('Error: '+res.error,'err'); }
    })
    .withFailureHandler(function(e){ hideLoader(); toast('Error: '+e.message,'err'); })
    .editarProducto({
      sku: sku,
      categoria: p.categoria,
      modelo: p.modelo,
      talla: p.talla,
      precioDetal: p.precioDetal,
      precioMayor: p.precioMayor,
      precioCosto: costoTotal
    });
}

function guardarReceta(){
  var sku    = document.getElementById('f-sku').value.trim();
  var modelo = document.getElementById('f-modelo').value.trim();
  var tela   = (document.getElementById('f-tela').value||'').trim().toUpperCase();
  var metros = parseFloat(document.getElementById('f-metros').value)||0;

  if(!sku && !modelo){ toast('Selecciona un producto del inventario primero','err'); document.getElementById('sku-search').focus(); return; }
  if(!tela){ toast('El tipo de tela es requerido','err'); document.getElementById('f-tela').focus(); return; }
  if(metros<=0){ toast('Los metros por prenda deben ser mayores a 0','err'); document.getElementById('f-metros').focus(); return; }

  var colorTipo = document.getElementById('f-color-tipo').value;
  var colorTela = colorTipo==='FIJO'
    ? (document.getElementById('f-color-fijo').value.trim().toUpperCase()||'SEGUN_PRENDA')
    : 'SEGUN_PRENDA';

  // Si tenemos SKU real, usamos el modelo del producto para coincidir con comandas
  var modeloReceta = modelo || sku;

  var datos = {
    modelo:          modeloReceta.toUpperCase(),
    talla:           document.getElementById('f-talla-receta').value || 'TODAS',
    tela:            tela,
    colorTela:       colorTela,
    metrosPorPrenda: metros,
    anchoTela:       parseFloat(document.getElementById('f-ancho').value)||0,
    observaciones:   (document.getElementById('f-obs').value||'').trim(),
    // Campos extra para referencia
    skuRef:          sku,
    precioMetro:     parseFloat(document.getElementById('f-precio-metro').value)||0,
    otrosCostos:     parseFloat(document.getElementById('f-otros-costos').value)||0
  };

  var btn = document.getElementById('btn-guardar');
  btn.disabled=true; btn.textContent='⏳ Guardando...';
  showLoader('Guardando receta...');

  google.script.run
    .withSuccessHandler(function(resp){
      hideLoader(); btn.disabled=false; btn.textContent='✓ Guardar Receta';
      if(!resp||!resp.ok){ toast('Error: '+(resp&&resp.error||'?'),'err'); return; }
      toast('✓ Receta guardada para '+modeloReceta,'ok');
      limpiarFormReceta();
      cargarDatos();
      switchTab('recetas');
    })
    .withFailureHandler(function(e){
      hideLoader(); btn.disabled=false; btn.textContent='✓ Guardar Receta';
      toast('Error: '+e.message,'err');
    })
    .agregarReceta(datos);
}

function filtrarRecetas(q){
  if(!q){ renderRecetas(recetas); return; }
  var ql=q.toLowerCase();
  renderRecetas(recetas.filter(function(r){
    return r.modelo.toLowerCase().indexOf(ql)!==-1 ||
           r.tela.toLowerCase().indexOf(ql)!==-1 ||
           (r.colorTela||'').toLowerCase().indexOf(ql)!==-1 ||
           (r.skuRef||'').toLowerCase().indexOf(ql)!==-1;
  }));
}

function renderRecetas(lista){
  var tbody = document.getElementById('recetas-tbody');
  if(!lista.length){
    tbody.innerHTML='<tr><td colspan="12" class="tbl-empty">No hay recetas. Agrega la primera en la pestaña "+ Nueva Receta"</td></tr>';
    return;
  }
  tbody.innerHTML = lista.map(function(r, i){
    var realIdx = recetas.indexOf(r);
    var colorChip = r.colorTela==='SEGUN_PRENDA'
      ? '<span class="chip chip-blue">Según prenda</span>'
      : '<span class="chip chip-green" style="display:inline-flex;align-items:center;gap:4px">'+cdotHtml(r.colorTela)+esc(r.colorTela)+'</span>';
    var tallaChip = r.talla==='TODAS'
      ? '<span class="chip chip-gray">Todas</span>'
      : '<span class="chip chip-gray">'+esc(r.talla)+'</span>';

    // Costo estimado si hay precio metro
    var costoEst = (r.precioMetro>0 && r.metrosPorPrenda>0)
      ? fmt(r.metrosPorPrenda*r.precioMetro + (r.otrosCostos||0))
      : '—';

    // SKU de referencia
    var skuCell = r.skuRef
      ? '<span class="td-sku">'+esc(r.skuRef)+'</span>'
      : '<span class="td-muted">—</span>';

    return '<tr>'+
      '<td>'+skuCell+'</td>'+
      '<td><strong>'+esc(r.modelo)+'</strong></td>'+
      '<td>'+
        (r.skuRef ? (function(){
          var p=productos.find(function(x){return x.sku===r.skuRef;});
          return p ? cdotHtml(p.color)+esc(p.color) : '<span class="td-muted">—</span>';
        })() : '<span class="td-muted">—</span>')+
      '</td>'+
      '<td>'+tallaChip+'</td>'+
      '<td><span class="chip chip-gold">'+esc(r.tela)+'</span></td>'+
      '<td>'+colorChip+'</td>'+
      '<td class="td-num">'+Number(r.metrosPorPrenda||0).toFixed(2)+' m</td>'+
      '<td class="td-num td-muted">'+(r.anchoTela||'—')+'</td>'+
      '<td class="td-num">'+(r.precioMetro>0?fmt(r.precioMetro):'—')+'</td>'+
      '<td class="td-num" style="color:var(--green)">'+costoEst+'</td>'+
      '<td class="td-muted">'+esc(r.observaciones||'—')+'</td>'+
      '<td>'+
        '<div style="display:flex;gap:5px">'+
          '<button class="btn btn-outline btn-sm" onclick="editarReceta('+realIdx+')">✎</button>'+
          '<button class="btn btn-red btn-sm" onclick="eliminarReceta('+realIdx+')">✕</button>'+
        '</div>'+
      '</td>'+
    '</tr>';
  }).join('');
}

function editarReceta(idx){
  var r = recetas[idx]; if(!r) return;
  editandoIdx = idx;
  switchTab('nueva');
  document.getElementById('form-title').textContent = '✎ Editando: '+r.modelo+' / '+r.tela;
  document.getElementById('btn-guardar').textContent = '✓ Actualizar Receta';

  // Si tiene SKU de referencia, buscar y seleccionar el producto
  if(r.skuRef){
    var p = productos.find(function(x){ return x.sku===r.skuRef; });
    if(p){ seleccionarSku(p.sku); }
  } else {
    // Solo llenar modelo manualmente
    document.getElementById('f-modelo').value = r.modelo;
  }

  document.getElementById('f-tela').value  = r.tela;
  document.getElementById('f-talla-receta').value = r.talla || 'TODAS';
  var esFijo = r.colorTela !== 'SEGUN_PRENDA';
  document.getElementById('f-color-tipo').value = esFijo ? 'FIJO' : 'SEGUN_PRENDA';
  toggleColorFijo();
  document.getElementById('f-color-fijo').value = esFijo ? r.colorTela : '';
  document.getElementById('f-metros').value = r.metrosPorPrenda||'';
  document.getElementById('f-ancho').value  = r.anchoTela||'';
  document.getElementById('f-precio-metro').value = r.precioMetro||'';
  document.getElementById('f-otros-costos').value = r.otrosCostos||'';
  document.getElementById('f-obs').value    = r.observaciones||'';
  calcularCostoPreview();
  document.getElementById('form-receta-panel') && document.getElementById('form-receta-panel').scrollIntoView({behavior:'smooth'});
}

function eliminarReceta(idx){
  var r=recetas[idx]; if(!r) return;
  if(!confirm('¿Eliminar receta de "'+r.modelo+' / '+r.tela+'"?\nEsta acción no se puede deshacer.')) return;
  showLoader('Eliminando receta...');
  google.script.run
    .withSuccessHandler(function(resp){
      hideLoader();
      if(!resp||!resp.ok){ toast('Error: '+(resp&&resp.error||'?'),'err'); return; }
      toast('Receta eliminada','ok');
      cargarDatos();
    })
    .withFailureHandler(function(e){ hideLoader(); toast('Error: '+e.message,'err'); })
    .eliminarRecetaAPI(idx);
}

function generarPlan(){
  showLoader('Calculando plan de tela desde comandas en PRODUCCIÓN...');
  google.script.run
    .withSuccessHandler(function(resp){
      hideLoader();
      if(!resp||!resp.ok){ toast('Error: '+(resp&&resp.error||'?'),'err'); return; }
      planData = resp.data;
      renderPlan(planData);
      switchTab('plan');
      var cnt = (planData&&planData.resumen)?planData.resumen.length:0;
      document.getElementById('tc-plan').textContent = cnt+' telas';
    })
    .withFailureHandler(function(e){ hideLoader(); toast('Error: '+e.message,'err'); })
    .generarPlanTela();
}

function renderPlan(data){
  var cont = document.getElementById('plan-content');
  if(!data || !data.resumen || !data.resumen.length){
    var aviso = (data&&data.aviso)||'No hay comandas en estado PRODUCCIÓN o no hay recetas configuradas.';
    cont.innerHTML = '<div class="no-plan">'+
      '<div class="no-plan-ico">📦</div>'+
      '<div class="no-plan-title">'+esc(aviso)+'</div>'+
      '<div class="no-plan-sub">'+
        'Asegúrate de que haya comandas en estado <strong>PRODUCCIÓN</strong><br>'+
        'y que sus modelos tengan recetas configuradas.'+
      '</div>'+

      cont.innerHTML += renderSinReceta(data.sinReceta);
    }
    return;
  }

  var totalPrendas = data.totalPrendas || data.resumen.reduce(function(a,r){return a+r.totalPrendas;},0);
  var totalMetros  = data.totalMetros  || data.resumen.reduce(function(a,r){return a+r.metros;},0);
  var totalMerma   = data.totalMetrosMerma || (totalMetros*1.08);
  var fechaGen = (data.generadoEn ? new Date(data.generadoEn) : new Date())
    .toLocaleString('es-VE',{day:'2-digit',month:'short',hour:'2-digit',minute:'2-digit'});

  var html = '';

  // HERO
  html += '<div class="plan-hero">'+
    '<div class="ph-left">'+
      '<div class="ph-eyebrow">Generado · '+esc(fechaGen)+'</div>'+
      '<h2>Plan de Compra de Tela</h2>'+
      '<p>'+data.comandas+' comanda(s) activa(s) en PRODUCCIÓN · '+data.resumen.length+' tipo(s) de tela necesarios</p>'+
    '</div>'+
    '<div class="ph-stats">'+
      '<div class="ph-stat"><div class="ph-stat-val">'+data.comandas+'</div><div class="ph-stat-lbl">Comandas</div></div>'+

      '<div class="ph-stat"><div class="ph-stat-val" style="color:#f59e0b">'+Number(totalMerma).toFixed(1)+'m</div><div class="ph-stat-lbl">+8% merma</div></div>'+
    '</div>'+
  '</div>';

  // TOTALES STRIP
  html += '<div class="tot-strip">';
  data.resumen.forEach(function(r){
    var colorLabel = (r.color||r.colorTela)==='SEGUN_PRENDA'?'multicolor':(r.color||r.colorTela||'');
    var metrosConMerma = r.metrosConMerma || (r.metros*1.08);
    html += '<div class="tot-item">'+
      '<div class="tot-val">'+Number(r.metros||0).toFixed(2)+'m</div>'+
      '<div class="tot-lbl">'+esc(r.tela)+(colorLabel?' · '+esc(colorLabel):'')+'</div>'+
      '<div style="font-family:\'DM Mono\',monospace;font-size:9px;color:var(--warn)">'+Number(metrosConMerma).toFixed(2)+'m c/merma</div>'+
    '</div>';
  });
  html += '</div>';

  // AVISO MERMA
  html += '<div style="display:flex;align-items:center;gap:12px;padding:10px 16px;background:#fff8ed;border:1px solid rgba(154,71,0,.2);border-left:3px solid var(--warn);margin-bottom:20px;font-size:11px;color:#6a3500">'+
    '<span style="font-size:15px">⚠️</span>'+
    '<span><strong>Merma textil del 8%</strong> incluida en los metros recomendados de compra. Representa el desperdicio promedio de corte.</span>'+
  '</div>';

  // TARJETAS DE TELA
  html += '<div class="telas-grid">';
  data.resumen.forEach(function(r, i){
    var cardId = 'det-'+i;
    var colorLabel = (r.color||r.colorTela)==='SEGUN_PRENDA' ? 'Según prenda' : (r.color||r.colorTela||'—');
    var colorDotColor = (r.color||r.colorTela)==='SEGUN_PRENDA' ? '#c9a84c' : colorHex(r.color||r.colorTela);
    var metrosConMerma = r.metrosConMerma || Math.round(r.metros*1.08*100)/100;
    var detalle = r.detalle || [];

    html += '<div class="tela-card">'+
      '<div class="tela-accent"></div>'+
      '<div class="tela-head">'+
        '<div class="tela-head-l">'+
          '<div class="tela-nombre">'+esc(r.tela)+'</div>'+
          '<div class="tela-color-row">'+
            '<span class="cdot" style="background:'+colorDotColor+'"></span>'+
            '<span class="tela-color-lbl">'+esc(colorLabel)+'</span>'+
          '</div>'+
        '</div>'+
        '<div class="tela-head-r">'+
          '<div class="tela-metros-val">'+Number(r.metros||0).toFixed(2)+'</div>'+
          '<div class="tela-metros-lbl">metros netos</div>'+
          '<div class="tela-merma">+merma: '+Number(metrosConMerma).toFixed(2)+'m</div>'+
        '</div>'+
      '</div>'+
      '<div class="tela-stats-row">'+
        '<div class="tela-stat"><div class="ts-val">'+r.totalPrendas+'</div><div class="ts-lbl">Prendas</div></div>'+
        '<div class="tela-stat"><div class="ts-val">'+detalle.length+'</div><div class="ts-lbl">Modelos</div></div>'+

      '</div>';

    if(detalle.length){
      html += '<button class="desglose-toggle" id="btn-'+cardId+'" onclick="toggleDesglose(\''+cardId+'\')">'+
        'Desglose por modelo y color <span class="dt-icon">▾</span>'+
      '</button>'+
      '<div class="desglose-body" id="'+cardId+'">'+
        '<div class="det-hdr">'+
          '<span>Modelo / Color</span>'+
          '<span style="text-align:center">Talla</span>'+
          '<span style="text-align:right">Prendas</span>'+
          '<span style="text-align:right">Metros</span>'+
        '</div>'+
        detalle.map(function(d){
          return '<div class="det-row">'+
            '<div>'+
              '<div class="det-modelo">'+esc(d.modelo)+'</div>'+
              '<div class="det-meta">'+cdotHtml(d.colorPrenda||'')+esc(d.colorPrenda||d.color||'—')+'</div>'+
            '</div>'+
            '<div class="det-talla">'+esc(d.talla||'UNICA')+'</div>'+
            '<div class="det-prendas">'+d.prendas+' uds</div>'+
            '<div class="det-metros">'+Number(d.metros||0).toFixed(2)+'m</div>'+
          '</div>';
        }).join('')+
      '</div>';
    }

    html += '</div>'; // tela-card
  });
  html += '</div>'; // telas-grid

  // SIN RECETA
  if(data.sinReceta && data.sinReceta.length){
    html += renderSinReceta(data.sinReceta);
  }

  cont.innerHTML = html;
}

function toggleDesglose(id){
  var body = document.getElementById(id);
  var btn  = document.getElementById('btn-'+id);
  var open = body.classList.contains('open');
  body.classList.toggle('open',!open);
  btn.classList.toggle('open',!open);
}

function showLoader(msg){
  document.getElementById('loader-msg').textContent = msg||'Procesando...';
  document.getElementById('loader').classList.add('show');
}
function hideLoader(){
  document.getElementById('loader').classList.remove('show');
}
function toast(msg, tipo){
  tipo = tipo||'ok';
  var t = document.createElement('div');
  t.className='toast '+(tipo==='ok'?'ok':tipo==='err'?'err':'');
  t.textContent = msg;
  document.getElementById('toast-ct').appendChild(t);
  setTimeout(function(){ t.style.opacity='0'; t.style.transition='opacity .3s'; setTimeout(function(){t.remove();},300); },3000);
}
