function colorDot(n) {

  var h = COLOR_MAP[k] || COLOR_MAP[k.split(' ')[0]] || '#9ca3af';
  return '<span class="color-dot" style="background:'+h+'"></span>';

function fmt(n)  { return '$' + Number(n).toFixed(2); }
function fmtF(d) { if (!d) return '—'; var p = d.split('-'); return p[2]+'/'+p[1]+'/'+p[0]; }
function estD(n) { if (n <= 0) return 'zero'; if (n <= 3) return 'low'; return 'ok'; }
function badgeD(n) {
  var e = estD(n);
  if (e==='zero') return '<span class="badge badge-empty">Sin stock</span>';
  if (e==='low')  return '<span class="badge badge-low">Stock bajo</span>';
  return '<span class="badge badge-ok">En stock</span>';
}
function margenPct(m, p) { return p > 0 ? Math.round((m/p)*100) : 0; }
function calcGanancia(p, qty) {
  if (!p || p.precioCosto <= 0) return null;
  return (p.precioDetal - p.precioCosto) * (qty || 1);
}
function calcGananciaM(m) {
  var p = productos.find(function(x){ return x.sku === m.sku; });
  if (!p || p.precioCosto <= 0 || m.tipo !== 'SALIDA') return null;
  var precio = m.precioVenta > 0 ? m.precioVenta : p.precioDetal;
  return (precio - p.precioCosto) * m.cantidad;
}
function gananciaBadge(g) {
  if (g === null) return '<span style="color:#666666;font-family:\'DM Mono\',monospace;font-size:10px">—</span>';
  var cls = g >= 0 ? 'var(--green)' : 'var(--red)';
  return '<span style="color:'+cls+';font-family:\'DM Mono\',monospace;font-size:11px;font-weight:700">'+(g>=0?'+':'')+fmt(g)+'</span>';
}
function marginStripHTML(colores) {
  var con = colores.filter(function(c){ return c.precioCosto > 0; });
  if (!con.length) return '';
  var c = con[0];
  var md = c.precioDetal - c.precioCosto;
  var mm = c.precioMayor > 0 ? c.precioMayor - c.precioCosto : null;
  var cd = md > 0 ? 'pos' : md < 0 ? 'neg' : 'zero';
  var cm = mm === null ? '' : mm > 0 ? 'pos' : mm < 0 ? 'neg' : 'zero';
  var html = '<div class="ganancia-strip">🔒 <span class="ganancia-chip">Costo: '+fmt(c.precioCosto)+'</span>'+
    '<span class="ganancia-chip '+cd+'">Margen detal: '+(md>=0?'+':'')+fmt(md)+' ('+margenPct(md,c.precioDetal)+'%)</span>';
  if (mm !== null) html += '<span class="ganancia-chip '+cm+'">Mayor: '+(mm>=0?'+':'')+fmt(mm)+' ('+margenPct(mm,c.precioMayor)+'%)</span>';
  html += '</div>';
  return html;
}
function agrupar(prods) {
  var map = {}, order = [];
  prods.forEach(function(p) {
    var key = p.categoria.trim().toLowerCase()+'||'+p.modelo.trim().toLowerCase();
    if (!map[key]) { map[key]={categoria:p.categoria,modelo:p.modelo,precioDetal:p.precioDetal,precioMayor:p.precioMayor,colores:[]}; order.push(key); }
    map[key].colores.push(p);
  });
  return order.map(function(k){ return map[k]; });
}

function setView(v) {
  currentView = v;
  ['group','card','list'].forEach(function(x){
    document.getElementById('view-'+x).style.display = x===v?'block':'none';
    document.getElementById('vbtn-'+x).classList.toggle('active',x===v);
  });
  renderInventario();
}
function getFilteredProds() {
  var q   = (document.getElementById('inv-search')?document.getElementById('inv-search').value:'').toLowerCase();
  var cat = document.getElementById('inv-cat')?document.getElementById('inv-cat').value:'';
  var est = document.getElementById('inv-estado')?document.getElementById('inv-estado').value:'';
  return productos.filter(function(p){
    return (!q || p.sku.toLowerCase().indexOf(q)!==-1 || p.modelo.toLowerCase().indexOf(q)!==-1 || p.color.toLowerCase().indexOf(q)!==-1)
      && (!cat || p.categoria===cat)
      && (!est || estD(p.disponible)===est);
  }).sort(function(a,b){
    var va=a[sortKey], vb=b[sortKey];
    return (typeof va==='string'?va.localeCompare(vb):va-vb)*sortDir;
  });
}
function sortBy(key) { if (sortKey===key) sortDir*=-1; else { sortKey=key; sortDir=1; } renderInventario(); }
function esMobile() {
  return window.innerWidth < 768;
}

}

function renderAlertas() {
  var criticos = productos.filter(function(p){ return p.disponible===0; });
  var bajos    = productos.filter(function(p){ return p.disponible>0 && p.disponible<=3; });
  var html = '';

    if (!items.length) return '';
    return '<div style="font-family:\'DM Mono\',monospace;font-size:9px;color:'+color+';letter-spacing:.14em;text-transform:uppercase;margin-bottom:8px;margin-top:14px">'+icon+' '+label+' ('+items.length+')</div>'+
      items.map(function(p){
        return '<div class="alert-item"  style="border-left-color:'+color+'">'+
          '<span style="font-size:18px">'+icon+'</span>'+
          '<div style="flex:1"><div style="font-size:13px;font-weight:600">'+p.modelo+' — '+colorDot(p.color)+' '+p.color+'</div><div style="font-size:10px;font-family:\'DM Mono\',monospace;color:#666666">'+p.sku+' · '+p.categoria+'</div></div>'+

          '<button class="btn btn-sm" style="background:'+color+';color:#fff" onclick="openModalMov(\''+p.sku+'\')">+ Entrada</button>'+
          '</div>';
      }).join('');
  }
  html += section(criticos,'var(--red)','Sin stock','⛔');
  html += section(bajos,'var(--warn)','Stock bajo','⚠️');
  if (!criticos.length && !bajos.length) html = '<div style="text-align:center;padding:60px;background:var(--surface);border:1px solid var(--border)"><div style="font-size:36px;margin-bottom:12px">✅</div><div style="color:var(--green);font-family:\'DM Mono\',monospace;font-size:12px">¡Todo el inventario está bien abastecido!</div></div>';

function onSkuModal() {

  document.getElementById('modal-p-color').textContent  = p.color;
  document.getElementById('modal-p-stock').textContent  = p.disponible;

function registrarMov() {
  var sku = document.getElementById('modal-sku').value;
  var fecha = document.getElementById('modal-fecha').value;
  var cantidad = parseInt(document.getElementById('modal-cantidad').value);
  var concepto = document.getElementById('modal-concepto').value;
  var contacto = document.getElementById('modal-contacto').value;
  var ref = document.getElementById('modal-ref').value;
  var tipo = modalTipo;
  if (!sku) { toast('Selecciona un producto','error'); return; }
  if (!cantidad || cantidad<1) { toast('Cantidad inválida','error'); return; }
  if (tipo==='SALIDA') {
    var p = productos.find(function(x){ return x.sku===sku; });
    if (p && p.disponible<cantidad) { toast('Stock insuficiente. Disponible: '+p.disponible,'error'); return; }
  }
  var btn = document.getElementById('modal-btn');
  btn.disabled=true; btn.textContent='⏳ Guardando...';
  google.script.run
    .withSuccessHandler(function(res){
      btn.disabled=false; btn.textContent='✓ Guardar';
      if (res.ok) { toast('✓ '+(tipo==='ENTRADA'?'Entrada':'Salida')+' de '+cantidad+' uds guardada','success'); closeMov(); cargarDatos(function(){ var pag=(document.querySelector('.page.active')||{}).id; if(pag) renderPage(pag.replace('page-','')); badge(); }); }
      else toast('Error: '+res.error,'error');

function buscarPorBarcode(prefix, codigo) {
  codigo = (codigo || '').trim();
  var msgEl   = document.getElementById(prefix + '-barcode-msg');
  var inputEl = document.getElementById(prefix + '-barcode');
  function devolverFoco() {
    requestAnimationFrame(function() {
      requestAnimationFrame(function() {
        if (inputEl) { inputEl.focus(); inputEl.select(); }
      });
    });
  }
  function showMsg(tipo, txt) {
    if (!msgEl) return;
    msgEl.className = 'barcode-msg ' + tipo;
    msgEl.textContent = txt;
    msgEl.style.display = 'block';
    setTimeout(function() { if (msgEl) msgEl.style.display = 'none'; }, 2600);
  }
  if (!codigo) { devolverFoco(); return; }
  var found = productos.find(function(p) {
    return p.sku === codigo || (p.codigoBarra && String(p.codigoBarra).trim() === codigo);
  });
  if (!found) {
    showMsg('error', '✕ No encontrado: ' + codigo);
    inputEl.value = '';
    devolverFoco();
    return;
  }
  if (prefix === 'sal' && found.disponible <= 0) {
    showMsg('warn', '⚠ Sin stock: ' + found.modelo + ' — ' + found.color);
    inputEl.value = '';
    devolverFoco();
    return;
  }
  inputEl.value = '';
  addToCart(prefix, found.sku);
  showMsg('ok', '✓ +1 · ' + found.modelo + ' — ' + found.color);
  devolverFoco();
}

function addToCart(prefix, sku) {
  var p=productos.find(function(x){ return x.sku===sku; }); if(!p) return;
  var existing=carts[prefix].find(function(x){ return x.sku===sku; });
  if (existing) {
    if (prefix==='sal'&&existing.qty>=p.disponible) { toast('Stock máximo: '+p.disponible,'error'); return; }
    existing.qty++;
  } else {
    var precio=prefix==='sal'?(salTipoVenta==='MAYOR'?p.precioMayor:p.precioDetal):p.precioDetal;
    carts[prefix].push({sku:p.sku,modelo:p.modelo,color:p.color,talla:p.talla,categoria:p.categoria,precio:precio,qty:1,disponible:p.disponible,precioMayor:p.precioMayor,precioDetal:p.precioDetal,precioCosto:p.precioCosto||0});
  }
  var searchEl = document.getElementById(prefix+'-search');
  if (searchEl) searchEl.value = '';
  var ddEl = document.getElementById(prefix+'-search-dd');
  if (ddEl) ddEl.style.display = 'none';
  renderCart(prefix);
}
function removeFromCart(prefix, sku) { carts[prefix]=carts[prefix].filter(function(x){ return x.sku!==sku; }); renderCart(prefix); }
function changeQty(prefix, sku, delta) {
  var item=carts[prefix].find(function(x){ return x.sku===sku; }); if(!item) return;
  var nq=item.qty+delta;
  if (nq<1) { removeFromCart(prefix,sku); return; }
  if (prefix==='sal'&&nq>item.disponible) { toast('Stock disponible: '+item.disponible,'error'); return; }
  item.qty=nq; renderCart(prefix);
}
function clearCart(prefix) { carts[prefix]=[]; renderCart(prefix); }
function actualizarCantDirecta(prefix, sku, inputEl) {
  var nq   = parseInt(inputEl.value, 10);
  var item = carts[prefix].find(function(x){ return x.sku === sku; });
  if (!item) return;
  if (isNaN(nq) || nq < 1) nq = 1;
  if (prefix === 'sal' && nq > item.disponible) {
    toast('Stock disponible: ' + item.disponible + ' uds', 'error');
    nq = item.disponible;
  }
  item.qty      = nq;
  inputEl.value = nq;
  renderCart(prefix);
}
function renderCart(prefix) {
  var items = carts[prefix], isSal = prefix === 'sal';
  if (isSal) items.forEach(function(i){ i.precio = salTipoVenta === 'MAYOR' ? i.precioMayor : i.precioDetal; });
  var container = document.getElementById(prefix + '-cart-items');
  var countEl   = document.getElementById(prefix + '-cart-count');
  var totalItems = items.reduce(function(a, i){ return a + i.qty; }, 0);
  if (countEl) countEl.textContent = items.length + ' producto' + (items.length !== 1 ? 's' : '') + ' · ' + totalItems + ' uds';
  container.innerHTML = !items.length
    ? '<div class="cart-empty">Busca o escanea un producto arriba 👆</div>'
    : items.map(function(item) {
        var sub = item.precio * item.qty;
        var ganChip = '';
        if (isSal && item.precioCosto > 0) {
          var gan = (item.precio - item.precioCosto) * item.qty;

        }
        return '<div class="cart-item">' +
          '<div class="ci-info">' +
            '<div class="ci-name">' + item.modelo + ' ' + colorDot(item.color) + ' ' + item.color + '</div>' +
            '<div class="ci-sku">'  + item.sku + ' · ' + item.talla + '</div>' +

              (isSal ? ' <span style="font-size:9px;color:#666666">· Stock: ' + item.disponible + '</span>' : '') +

      clienteId:  clienteId
    };
  });

  google.script.run
    .withSuccessHandler(function(res){
      btn.disabled=false;
      btn.textContent=tipo==='ENTRADA'?'✓ Registrar Entrada':'✓ Registrar Salida';
      if (!res||!res.ok) {
        toast('Error: '+(res&&res.error?res.error:'Error desconocido'),'error');
        return;
      }
      var uds=snap.reduce(function(a,i){ return a+i.qty; },0);
      var monto=snap.reduce(function(a,i){ return a+(i.precio*i.qty); },0);
      toast(tipo==='SALIDA'?'✓ Venta de '+uds+' uds por '+fmt(monto):'✓ Entrada de '+uds+' uds','success');
      clearCart(prefix);
      ['concepto','ref'].forEach(function(f){ var e=document.getElementById(prefix+'-'+f); if(e) e.value=''; });
      cargarDatos(function(){
        badge();
        var pag=(document.querySelector('.page.active')||{}).id;
        if (pag) renderPage(pag.replace('page-',''));
      });
    })
    .withFailureHandler(function(e){
      btn.disabled=false;
      btn.textContent=tipo==='ENTRADA'?'✓ Registrar Entrada':'✓ Registrar Salida';
      toast('Error: '+e.message,'error');
    })
    .registrarMovimientos(lote);
}

function exportCSV() {

function exportHisCSV() {
  var rows=[['ID','FECHA','SKU','CATEGORIA','MODELO','COLOR','TIPO','CANTIDAD','TIPO_VENTA','PRECIO_VENTA','GANANCIA_BRUTA','CONCEPTO','CLIENTE']];
  getFilteredMovs().forEach(function(m){
    var p=productos.find(function(x){ return x.sku===m.sku; });
    var g=calcGananciaM(m);
    rows.push([m.id,m.fecha,m.sku,p?p.categoria:'',p?p.modelo:'',p?p.color:'',m.tipo,m.cantidad,m.tipoVenta||'',m.precioVenta||0,g!==null?g:'',m.concepto,m.contacto]);
  });
  dlCSV(rows,'movimientos_'+new Date().toISOString().split('T')[0]+'.csv');
  toast('Historial exportado','success');
}
function dlCSV(rows, name) {
  var csv=rows.map(function(r){ return r.map(function(v){ return '"'+String(v).replace(/"/g,'""')+'"'; }).join(','); }).join('\n');
  var a=document.createElement('a');
  a.href=URL.createObjectURL(new Blob(['\uFEFF'+csv],{type:'text/csv;charset=utf-8'}));
  a.download=name; a.click();
}

function renderPag(cid, cur, total, onClick) {
  var c=document.getElementById(cid); if(!c) return;
  var h='<button class="page-btn" onclick="('+onClick.toString()+')('+(cur-1)+')" '+(cur===1?'disabled':'')+'>‹</button>';
  for (var i=1;i<=total;i++) {
    if (total>7&&i>2&&i<total-1&&Math.abs(i-cur)>1) { if(i===3||i===total-2) h+='<span style="padding:0 4px;color:#666666">…</span>'; continue; }
    h+='<button class="page-btn '+(i===cur?'active':'')+'" onclick="('+onClick.toString()+')('+i+')">'+i+'</button>';
  }
  h+='<button class="page-btn" onclick="('+onClick.toString()+')('+(cur+1)+')" '+(cur===total?'disabled':'')+'>›</button>';
  c.innerHTML=h;
}

function abrirEnvios() {
  var nombreEl   = document.getElementById('sal-contacto');
  var clienteId  = (document.getElementById('sal-cliente-id')||{}).value||'';
  var refEl      = document.getElementById('sal-ref');
  var conceptoEl = document.getElementById('sal-concepto');
  var nombre = (nombreEl   ? nombreEl.value   : '').trim();
  var ref    = (refEl      ? refEl.value      : '').trim();
  var nota   = (conceptoEl ? conceptoEl.value : '').trim();
  var tel = '', ciudad = '';
  if (clienteId && typeof clientes !== 'undefined') {
    var cli = clientes.find(function(c){ return c.id === clienteId; });
    if (cli) { nombre = cli.nombre||nombre; tel=cli.telefono||''; ciudad=cli.ciudad||''; }
  }
  var destinatarios = [];
  if (nombre) destinatarios.push({nombre:nombre,telefono:tel,agencia:'',ciudad:ciudad,ref:ref,nota:nota});
  var dataParam = destinatarios.length ? encodeURIComponent(JSON.stringify(destinatarios)) : '';
  google.script.run
    .withSuccessHandler(function(baseUrl){ var url=dataParam?baseUrl+'&data='+dataParam:baseUrl; window.open(url,'_blank'); })
    .withFailureHandler(function(err){ toast('Error al abrir envíos: '+(err.message||err),'error'); })
    .getEnviosUrl();
}

function cargarPlanTela() {

    .withSuccessHandler(function(res){

    .withFailureHandler(function(e){ _ptError(e.message||String(e)); })

function renderPlanTela(data) {
  document.getElementById('pt-loading').style.display = 'none';
  if (!data||!data.resumen||!data.resumen.length) {
    document.getElementById('pt-empty').style.display = 'block';
    document.getElementById('pt-empty-msg').textContent = (data&&data.aviso)?data.aviso:'No hay comandas en estado PRODUCCIÓN.';
    _ptRenderSinReceta(data&&data.sinReceta?data.sinReceta:[]);
    return;
  }
  document.getElementById('pt-kpis').style.display       = 'block';
  document.getElementById('pt-table-wrap').style.display = 'block';
  document.getElementById('pt-print-btn').style.display  = 'inline-flex';
  document.getElementById('pt-k-comandas').textContent = data.comandas||0;
  document.getElementById('pt-k-prendas').textContent  = data.totalPrendas||0;
  document.getElementById('pt-k-metros').textContent   = (data.totalMetros||0).toFixed(2)+' m';

  }

  var tbody=document.getElementById('pt-tbody');
  tbody.innerHTML=data.resumen.map(function(r,i){
    var hex=(function(n){ var k=(n||'').toUpperCase().trim(); return COLOR_MAP[k]||COLOR_MAP[k.split(' ')[0]]||'#aaaaaa'; })(r.color);
    var dot=r.color&&r.color!=='—'?'<span style="display:inline-block;width:9px;height:9px;border-radius:50%;background:'+hex+';border:1px solid rgba(0,0,0,.14);vertical-align:middle;margin-right:5px"></span>':'';
    return '<tr style="'+(i%2?'background:#fafafa':'')+'">'+
      '<td><strong style="font-family:\'Playfair Display\',serif">'+ptEsc(r.tela)+'</strong></td>'+
      '<td>'+dot+ptEsc(r.color)+'</td>'+
      '<td style="text-align:right;font-family:\'DM Mono\',monospace;font-weight:700">'+r.totalPrendas+' uds</td>'+
      '<td style="text-align:right;font-family:\'DM Mono\',monospace;font-weight:700">'+r.metros.toFixed(2)+' m</td>'+
      '<td style="text-align:right;font-family:\'DM Mono\',monospace;font-weight:700;color:var(--warn)">'+r.metrosConMerma.toFixed(2)+' m</td>'+
      '</tr>';
  }).join('');
  document.getElementById('pt-tfoot').innerHTML=
    '<tr style="background:#1a1714;color:#fff">'+
      '<td colspan="2" style="padding:11px 14px;font-family:\'DM Mono\',monospace;font-size:9px;letter-spacing:.14em;text-transform:uppercase">TOTALES</td>'+
      '<td style="text-align:right;font-family:\'DM Mono\',monospace;font-weight:700;padding:11px 14px">'+(data.totalPrendas||0)+' uds</td>'+
      '<td style="text-align:right;font-family:\'DM Mono\',monospace;font-weight:700;padding:11px 14px">'+(data.totalMetros||0).toFixed(2)+' m</td>'+

function _ptRenderSinReceta(sinReceta) {
  var sec=document.getElementById('pt-sin-receta');
  if (!sinReceta||!sinReceta.length) { sec.style.display='none'; return; }
  sec.style.display='block';
  function ptEsc(s){ return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }
  document.getElementById('pt-sr-tbody').innerHTML=sinReceta.map(function(p){
    return '<tr><td>'+ptEsc(p.modelo)+'</td><td>'+ptEsc(p.color||'—')+'</td><td>'+ptEsc(p.talla||'—')+'</td><td style="text-align:right;font-family:\'DM Mono\',monospace;font-weight:700">'+p.totalPrendas+'</td></tr>';
  }).join('');
}

(function initBarcodeListeners() {

    { inputId:'ent-barcode', clearId:'ent-barcode-clear', prefix:'ent' },
    { inputId:'sal-barcode', clearId:'sal-barcode-clear', prefix:'sal' }
  ];
  pairs.forEach(function(pair) {
    var inputEl=document.getElementById(pair.inputId);
    var clearEl=document.getElementById(pair.clearId);
    if (!inputEl) return;
    inputEl.addEventListener('keydown', function(e){
      if (e.key==='Enter') {
        e.preventDefault(); e.stopPropagation();
        var codigo=inputEl.value;
        inputEl.value='';
        buscarPorBarcode(pair.prefix, codigo);
      }
    });
    if (clearEl) {
      clearEl.addEventListener('mousedown', function(e){ e.preventDefault(); });
      clearEl.addEventListener('click', function(){
        inputEl.value=''; inputEl.focus();
        var msgEl=document.getElementById(pair.prefix+'-barcode-msg');
        if (msgEl) msgEl.style.display='none';
      });
    }
  });
})();

<script>
// ════════════════════════════════════════════════════════
//  MODITEX — PRODUCT PICKER  v2.0
//  Selector visual de productos por Categoría › Modelo › Variante
//  Integrado en Index.html. Usa la variable global `productos`.
// ════════════════════════════════════════════════════════

var _picker = {
  paso:         1,        // 1=cats, 2=modelos, 3=variantes, 4=búsqueda
  catActual:    null,
  modeloActual: null,
  callback:     null,     // fn(sku) al seleccionar variante
  tipo:         'SALIDA', // 'ENTRADA' | 'SALIDA'
  busqueda:     ''
};

// Iconos por categoría
var _pickerIconos = {
  'CHAQUETA':'🧥','JACKET':'🧥','ABRIGO':'🧥',
  'PANTALON':'👖','JEAN':'👖','PANTS':'👖','JOGGER':'👖',
  'CAMISA':'👔','BLUSA':'👔','TOPS':'👔',
  'VESTIDO':'👗','FALDA':'👗','ENTERIZO':'👗',
  'CONJUNTO':'🩱','TRAJE':'🩱','TRIKINIS':'🩱',
  'ZAPATO':'👟','BOTA':'👢','ZAPATILLA':'👟',
  'SHORT':'🩳','BERMUDA':'🩳',
  'PIJAMA':'🛌','INTERIOR':'🩲',
};
function _pickerIcono(cat) {
  var c = (cat||'').toUpperCase();
  for (var k in _pickerIconos) { if (c.indexOf(k) !== -1) return _pickerIconos[k]; }
  return '🏷️';
}

// Colores hex para dot
function _pickerDotHex(color) {
  var k = (color||'').toUpperCase().trim();
  return COLOR_MAP[k] || COLOR_MAP[k.split(' ')[0]] || '#9ca3af';
}

function _pickerFmt(n) {
  return '$' + Number(n||0).toFixed(2);
}

// ── API pública ─────────────────────────────────────────────────

/**
 * Abre el picker.
 * @param {function} callback  fn(sku) al seleccionar
 * @param {string}   tipo      'ENTRADA' | 'SALIDA'
 */
function abrirPicker(callback, tipo) {
  _picker.callback     = callback || function(){};
  _picker.tipo         = (tipo || 'SALIDA').toUpperCase();
  _picker.paso         = 1;
  _picker.catActual    = null;
  _picker.modeloActual = null;
  _picker.busqueda     = '';

  document.getElementById('pickerSearchInput').value = '';
  document.getElementById('productPicker').classList.add('open');
  document.getElementById('pickerSearchBar').style.display = '';

  var badge = document.getElementById('pickerTipoBadge');
  badge.textContent = _picker.tipo;
  badge.className   = 'picker-tipo-badge ' + (_picker.tipo === 'ENTRADA' ? 'entrada' : 'salida');

  _pickerRenderCats();
}

function cerrarPicker() {
  document.getElementById('productPicker').classList.remove('open');
}

// ── Paso 1: Categorías ───────────────────────────────────────────

function _pickerRenderCats(busq) {
  _picker.paso = 1;
  _picker.catActual    = null;
  _picker.modeloActual = null;

  document.getElementById('pickerBreadcrumb').innerHTML =
    '<span class="bc-current">Categorías</span>';

  // Si hay búsqueda, mostrar resultados directos
  if (busq && busq.length >= 2) {
    _pickerRenderBusqueda(busq);
    return;
  }

  // Agrupar por categoría
  var cats = {};
  productos.forEach(function(p) {
    var cat = p.categoria || 'OTROS';
    if (!cats[cat]) cats[cat] = { total: 0, conStock: 0, stockTotal: 0 };
    cats[cat].total++;
    cats[cat].stockTotal += p.disponible;
    if (p.disponible > 0) cats[cat].conStock++;
  });

  var keys = Object.keys(cats).sort();
  if (!keys.length) {
    _pickerSetBody('<div class="picker-empty-state"><div class="es-icon">📦</div><div class="es-msg">No hay productos cargados</div></div>');
    return;
  }

  var html = '<p class="picker-section-label">Categorías (' + keys.length + ')</p>';
  html += '<div class="picker-grid-cat">';
  keys.forEach(function(cat) {
    var info = cats[cat];
    html += '<button class="btn-pcat" onclick="_pickerSelCat(\'' + _pickerEsc(cat) + '\')">'
      + '<span class="pcat-icon">' + _pickerIcono(cat) + '</span>'
      + '<span class="pcat-name">' + cat + '</span>'
      + '<span class="pcat-stock">' + info.conStock + ' con stock</span>'
      + '</button>';
  });
  html += '</div>';
  _pickerSetBody(html);
}

// ── Paso 2: Modelos ──────────────────────────────────────────────

function _pickerSelCat(cat) {
  _picker.paso      = 2;
  _picker.catActual = cat;

  document.getElementById('pickerBreadcrumb').innerHTML =
    '<span class="bc-item" onclick="_pickerRenderCats()">Categorías</span>'
    + '<span class="bc-sep">›</span>'
    + '<span class="bc-current">' + cat + '</span>';

  var prods = productos.filter(function(p) { return p.categoria === cat; });

  // Agrupar por modelo
  var modelos = {};
  prods.forEach(function(p) {
    var m = p.modelo || 'SIN MODELO';
    if (!modelos[m]) modelos[m] = { stockTotal: 0, conStock: 0, total: 0 };
    modelos[m].total++;
    modelos[m].stockTotal += p.disponible;
    if (p.disponible > 0) modelos[m].conStock++;
  });

  var keys = Object.keys(modelos).sort();
  var html = '<p class="picker-section-label">' + cat + ' — ' + keys.length + ' modelos</p>';
  html += '<div class="picker-grid-mod">';
  keys.forEach(function(mod) {
    var info = modelos[mod];
    var cls  = info.stockTotal === 0 ? 'pmod-none' : info.conStock < 3 ? 'pmod-low' : 'pmod-ok';
    var txt  = info.stockTotal === 0 ? 'Sin stock' : info.stockTotal + ' uds en stock';
    html += '<button class="btn-pmod" onclick="_pickerSelModelo(\'' + _pickerEsc(cat) + '\',\'' + _pickerEsc(mod) + '\')">'
      + '<div class="pmod-name">' + mod + '</div>'
      + '<div class="pmod-meta">' + info.total + ' variantes · <span class="' + cls + '">' + txt + '</span></div>'
      + '</button>';
  });
  html += '</div>';
  _pickerSetBody(html);
}

// ── Paso 3: Variantes ────────────────────────────────────────────

function _pickerSelModelo(cat, modelo) {
  _picker.paso         = 3;
  _picker.catActual    = cat;
  _picker.modeloActual = modelo;

  document.getElementById('pickerBreadcrumb').innerHTML =
    '<span class="bc-item" onclick="_pickerRenderCats()">Categorías</span>'
    + '<span class="bc-sep">›</span>'
    + '<span class="bc-item" onclick="_pickerSelCat(\'' + _pickerEsc(cat) + '\')">' + cat + '</span>'
    + '<span class="bc-sep">›</span>'
    + '<span class="bc-current">' + modelo + '</span>';

  // Ocultar búsqueda en paso 3 para mayor claridad
  document.getElementById('pickerSearchBar').style.display = 'none';

  var prods = productos.filter(function(p) {
    return p.categoria === cat && p.modelo === modelo;
  });

  // Ordenar: con stock primero, luego por talla
  prods = prods.slice().sort(function(a, b) {
    if (a.disponible > 0 && b.disponible === 0) return -1;
    if (a.disponible === 0 && b.disponible > 0) return 1;
    return (a.talla || '').localeCompare(b.talla || '');
  });

  var html = '<p class="picker-section-label">' + modelo + ' — ' + prods.length + ' variante(s)</p>';
  html += '<div class="picker-grid-var">';

  prods.forEach(function(p) {
    var badgeClass, badgeTxt;
    if (p.disponible === 0) {
      badgeClass = 'pvsb-none'; badgeTxt = 'Sin stock';
    } else if (p.disponible <= 3) {
      badgeClass = 'pvsb-low'; badgeTxt = p.disponible + ' uds';
    } else {
      badgeClass = 'pvsb-ok'; badgeTxt = p.disponible + ' uds';
    }

    var noStock   = p.disponible === 0 && _picker.tipo === 'SALIDA';
    var precio    = _picker.tipo === 'ENTRADA' ? p.precioCosto : p.precioDetal;
    var dotHex    = _pickerDotHex(p.color);
    var tallaHtml = (p.usaVariantes && p.talla && p.talla !== 'UNICA')
      ? '<div class="pvar-talla">Talla: ' + p.talla + '</div>'
      : '';

    html += '<button class="btn-pvar' + (noStock ? ' pvar-nostock' : '') + '"'
      + (noStock ? '' : ' onclick="_pickerSelVariante(\'' + _pickerEsc(p.sku) + '\')"') + '>'
      + '<span class="pvar-stock-badge ' + badgeClass + '">' + badgeTxt + '</span>'
      + '<div class="pvar-color-row">'
      + '<span class="color-dot" style="background:' + dotHex + '"></span>'
      + '<span class="pvar-color-name">' + (p.color || '—') + '</span>'
      + '</div>'
      + tallaHtml
      + '<div class="pvar-sku">' + p.sku + '</div>'
      + '<div class="pvar-precio">' + _pickerFmt(precio) + '</div>'
      + (noStock ? '<div class="pvar-sinstock-overlay">Sin stock</div>' : '')
      + '</button>';
  });

  html += '</div>';
  _pickerSetBody(html);
}

// ── Selección final ──────────────────────────────────────────────

function _pickerSelVariante(sku) {
  cerrarPicker();
  if (_picker.callback) _picker.callback(sku);
  toast('✓ ' + sku + ' agregado al carrito', 'success');
}

// ── Búsqueda global ──────────────────────────────────────────────

function filtrarPicker(val) {
  _picker.busqueda = (val || '').trim();
  if (_picker.busqueda.length < 2) {
    // Volver a la vista anterior
    if (_picker.catActual && _picker.modeloActual) {
      document.getElementById('pickerSearchBar').style.display = '';
      _pickerSelModelo(_picker.catActual, _picker.modeloActual);
    } else if (_picker.catActual) {
      _pickerSelCat(_picker.catActual);
    } else {
      _pickerRenderCats();
    }
    return;
  }
  document.getElementById('pickerSearchBar').style.display = '';
  _pickerRenderBusqueda(_picker.busqueda);
}

function _pickerRenderBusqueda(q) {
  _picker.paso = 4;
  var ql = q.toUpperCase();

  document.getElementById('pickerBreadcrumb').innerHTML =
    '<span class="bc-item" onclick="_pickerRenderCats()">Categorías</span>'
    + '<span class="bc-sep">›</span>'
    + '<span class="bc-current">Búsqueda: "' + q + '"</span>';

  var matches = productos.filter(function(p) {
    return (p.sku||'').toUpperCase().indexOf(ql) !== -1
        || (p.modelo||'').toUpperCase().indexOf(ql) !== -1
        || (p.color||'').toUpperCase().indexOf(ql) !== -1
        || (p.categoria||'').toUpperCase().indexOf(ql) !== -1;
  });

  if (!matches.length) {
    _pickerSetBody('<div class="picker-empty-state"><div class="es-icon">🔍</div><div class="es-msg">Sin resultados para "' + q + '"</div></div>');
    return;
  }

  // Ordenar: con stock primero
  matches = matches.slice().sort(function(a, b) {
    if (a.disponible > 0 && b.disponible === 0) return -1;
    if (a.disponible === 0 && b.disponible > 0) return 1;
    return (a.modelo||'').localeCompare(b.modelo||'');
  });

  var html = '<p class="picker-section-label">' + matches.length + ' resultado(s)</p>';
  html += '<div class="picker-grid-var">';

  matches.forEach(function(p) {
    var badgeClass, badgeTxt;
    if (p.disponible === 0) {
      badgeClass = 'pvsb-none'; badgeTxt = 'Sin stock';
    } else if (p.disponible <= 3) {
      badgeClass = 'pvsb-low'; badgeTxt = p.disponible + ' uds';
    } else {
      badgeClass = 'pvsb-ok'; badgeTxt = p.disponible + ' uds';
    }

    var noStock = p.disponible === 0 && _picker.tipo === 'SALIDA';
    var precio  = _picker.tipo === 'ENTRADA' ? p.precioCosto : p.precioDetal;
    var dotHex  = _pickerDotHex(p.color);

    html += '<button class="btn-pvar' + (noStock ? ' pvar-nostock' : '') + '"'
      + (noStock ? '' : ' onclick="_pickerSelVariante(\'' + _pickerEsc(p.sku) + '\')"') + '>'
      + '<span class="pvar-stock-badge ' + badgeClass + '">' + badgeTxt + '</span>'
      + '<div class="pvar-color-row">'
      + '<span class="color-dot" style="background:' + dotHex + '"></span>'
      + '<span class="pvar-color-name" style="font-size:11px">' + p.modelo + '</span>'
      + '</div>'
      + '<div style="font-size:10px;color:#666;margin:3px 0">' + (p.color||'') + (p.talla && p.talla!=='UNICA' ? ' · T:' + p.talla : '') + '</div>'
      + '<div class="pvar-sku">' + p.sku + '</div>'
      + '<div class="pvar-precio">' + _pickerFmt(precio) + '</div>'
      + (noStock ? '<div class="pvar-sinstock-overlay">Sin stock</div>' : '')
      + '</button>';
  });

  html += '</div>';
  _pickerSetBody(html);
}

<script>
// ── Bottom Nav: popup movimientos ──────────────────────────────
function toggleMovMenu(e) {
  e.stopPropagation();
  var popup = document.getElementById('mov-popup');
  popup.style.display = popup.style.display === 'none' ? 'block' : 'none';
}
function cerrarMovPopup() {
  var popup = document.getElementById('mov-popup');
  if (popup) popup.style.display = 'none';
}
document.addEventListener('click', function(e) {
  var popup = document.getElementById('mov-popup');
  if (popup && popup.style.display !== 'none') {
    if (!popup.contains(e.target) && e.target.id !== 'btn-mov-bottom') {
      cerrarMovPopup();
    }
  }
});
</script>
