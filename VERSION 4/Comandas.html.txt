// ── DRAG & DROP ────────────────────────────────────────────────
function initDnD(){
  document.querySelectorAll('.cmd-card[data-id]').forEach(function(card){
    card.setAttribute('draggable','true');
    card.addEventListener('dragstart',function(e){
      e.dataTransfer.setData('cmdId',card.dataset.id);
      e.dataTransfer.effectAllowed='move';
      setTimeout(function(){card.classList.add('dragging');},0);
    });
    card.addEventListener('dragend',function(){
      card.classList.remove('dragging');
      document.querySelectorAll('.col-cards').forEach(function(z){z.classList.remove('drag-over');});
    });
  });
  document.querySelectorAll('.col-cards').forEach(function(zone){
    zone.addEventListener('dragover',function(e){e.preventDefault();e.dataTransfer.dropEffect='move';zone.classList.add('drag-over');});
    zone.addEventListener('dragleave',function(e){if(!zone.contains(e.relatedTarget))zone.classList.remove('drag-over');});
    zone.addEventListener('drop',function(e){
      e.preventDefault();zone.classList.remove('drag-over');
      var cmdId=e.dataTransfer.getData('cmdId');
      if(!cmdId)return;
      var newStatus=zone.dataset.col;
      var cmd=comandas.find(function(c){return c.id===cmdId;});

    });
  });
}

  if(!productos.length){body.innerHTML='<div class="cat-no-results">⚠ Sin productos. Verifica la hoja Inventario.</div>';return;}
  if(busq){
    var m=productos.filter(function(p){return p.sku.toLowerCase().indexOf(busq)!==-1||p.modelo.toLowerCase().indexOf(busq)!==-1||p.color.toLowerCase().indexOf(busq)!==-1||p.categoria.toLowerCase().indexOf(busq)!==-1;});
    body.innerHTML='<div class="cat-sec-lbl">'+m.length+' resultado'+(m.length!==1?'s':'')+'</div>'+(m.length?renderProdsGrid(m):'<div class="cat-no-results">Sin resultados para "'+esc(busq)+'"</div>');
    return;
  }
  if(catVista==='categorias'){
    var cats={};
    productos.forEach(function(p){if(!cats[p.categoria])cats[p.categoria]={n:0,modelos:{}};cats[p.categoria].n++;cats[p.categoria].modelos[p.modelo]=true;});
    var names=Object.keys(cats).sort();
    if(!names.length){body.innerHTML='<div class="cat-no-results">Sin categorías. Revisa la hoja Inventario.</div>';return;}
    body.innerHTML='<div class="cat-sec-lbl">'+names.length+' categorías</div><div class="cat-cats-grid">'+
      names.map(function(cat){var nm=Object.keys(cats[cat].modelos).length;return '<div class="cat-cat-btn" onclick="verCategoria(\''+esc(cat)+'\')"><div class="cat-cat-icon">'+catIcon(cat)+'</div><div class="cat-cat-name">'+esc(cat)+'</div><div class="cat-cat-count">'+nm+' modelo'+(nm!==1?'s':'')+'</div></div>';}).join('')+'</div>';
    return;
  }
  if(catVista==='modelos'){
    var prodsAll=productos.filter(function(p){return p.categoria===catCat;});
    var mMap={};
    prodsAll.forEach(function(p){if(!mMap[p.modelo])mMap[p.modelo]={vars:[],stock:0,pd:[],pm:[]};mMap[p.modelo].vars.push(p);mMap[p.modelo].stock+=p.disponible||0;if(p.precioDetal>0)mMap[p.modelo].pd.push(p.precioDetal);if(p.precioMayor>0)mMap[p.modelo].pm.push(p.precioMayor);});
    var mNames=Object.keys(mMap).sort();
    body.innerHTML=breadcrumb()+'<div class="cat-sec-lbl" style="margin-top:4px">'+mNames.length+' modelos</div><div class="cat-modelos-grid">'+
      mNames.map(function(m){
        var info=mMap[m],hasStock=info.stock>0;
        var inCart=cmdCart.filter(function(x){return info.vars.some(function(v){return v.sku===x.sku;});}).reduce(function(a,x){return a+x.cant;},0);
        var cls='cat-modelo-btn'+(inCart?' in-cart':hasStock?' has-stock':'');
        var pdStr=info.pd.length?(Math.min.apply(null,info.pd)===Math.max.apply(null,info.pd)?fmtEur(info.pd[0]):fmtEur(Math.min.apply(null,info.pd))+'-'+fmtEur(Math.max.apply(null,info.pd))):'—';
        var pmStr=info.pm.length?(Math.min.apply(null,info.pm)===Math.max.apply(null,info.pm)?fmtEur(info.pm[0]):fmtEur(Math.min.apply(null,info.pm))+'-'+fmtEur(Math.max.apply(null,info.pm))):'—';
        return '<div class="'+cls+'" onclick="verModelo(\''+esc(m)+'\')"><span style="font-size:20px;flex-shrink:0">'+catIcon(catCat)+'</span><div class="cat-modelo-info"><div class="cat-modelo-name">'+esc(m)+'</div><div class="cat-modelo-meta">'+info.vars.length+' var. · '+(hasStock?'<span style="color:var(--done)">'+info.stock+' uds</span>':'Sin stock')+(inCart?'<span style="color:var(--prod);font-weight:700"> ✓ ×'+inCart+'</span>':'')+'</div><div class="cat-modelo-meta" style="margin-top:1px"><span style="color:var(--prod)">D:'+pdStr+'</span> <span style="color:var(--gold-d)">M:'+pmStr+'</span></div></div><span style="font-family:\'DM Mono\',monospace;font-size:10px;color:var(--muted2)">→</span></div>';
      }).join('')+'</div>';
    return;
  }
  if(catVista==='productos'){
    var prods2=productos.filter(function(p){return p.categoria===catCat&&p.modelo===catModelo;});
    body.innerHTML=breadcrumb()+'<div class="cat-sec-lbl" style="margin-top:4px">'+prods2.length+' variante'+(prods2.length!==1?'s':'')+'</div>'+renderProdsGrid(prods2);
  }
}

function guardarPedido(){
  if(isSaving){toast('Ya está guardando...','err');return;}
  var cartEls=document.querySelectorAll('.cart-item');
  cmdCart.forEach(function(item,idx){
    if(item.tipoSku==='UNICO')return;
    var el=cartEls[idx];if(!el)return;
    var tEl=el.querySelector('select'),cEl=el.querySelector('input[type="text"]');
    if(tEl)item.talla=tEl.value||'UNICA';if(cEl)item.color=(cEl.value||'').trim().toUpperCase();
  });
  var cliente=(document.getElementById('f-cliente').value||'').trim();
  if(!cliente){toast('Nombre del cliente requerido','err');document.getElementById('f-cliente').focus();return;}
  if(!cmdCart.length){toast('Agrega al menos una prenda','err');return;}
  var totalEur=cmdCart.reduce(function(s,i){var p=(i.tv||'DETAL')==='MAYOR'?i.precioMayor:i.precioDetal;return s+(p*i.cant);},0);
  var tasa=parseFloat((document.getElementById('f-tasa')||{}).value)||0;
  var tiposEnCart=[...new Set(cmdCart.map(function(i){return i.tv||'DETAL';}))];
  var tvTag=tiposEnCart.length>1?'MIXTO':tiposEnCart[0]||'DETAL';
  var notaU=(document.getElementById('f-nota').value||'').trim();
  var metaStr='['+tvTag+']';if(tasa>0){var bs2=totalEur*tasa;metaStr+=' EUR:'+tasa.toFixed(2)+' → Bs.'+bs2.toFixed(2);}
  var notaFinal=metaStr+(notaU?' | '+notaU:'');
  var pim=parseFloat(document.getElementById('f-pago-inicial').value)||0;
  var pimet=document.getElementById('f-pago-inicial-metodo').value||'';
  var pinota=document.getElementById('f-pago-inicial-nota').value||'';
  var datos={cliente:cliente,clienteId:document.getElementById('f-cliente-id').value||'',telefono:(document.getElementById('f-tel').value||'').trim(),
    productos:cmdCart.map(function(i){return{cant:i.cant,sku:i.sku,nombre:i.nombre,talla:i.talla||'',color:i.color||'',tv:i.tv||'DETAL'};}),
    precio:parseFloat(totalEur.toFixed(2)),pago:document.getElementById('f-pago').value,agencia:(document.getElementById('f-agencia').value||'').trim(),
    nota:notaFinal,montoPagadoInicial:pim,notaPagoInicial:[pimet,pinota].filter(Boolean).join(' · ')};
  var btn=document.getElementById('m-save-btn');
  isSaving=true;btn.disabled=true;btn.textContent='⏳ Guardando...';
  var onOk=function(res){isSaving=false;btn.disabled=false;btn.textContent='✓ Guardar Pedido';if(res.ok){cerrarModal();toast('Pedido '+(editandoId?'actualizado':'guardado')+' ✓');cargarTodo();}else toast('Error: '+res.error,'err');};
  var onErr=function(e){isSaving=false;btn.disabled=false;btn.textContent='✓ Guardar Pedido';toast('Error: '+e.message,'err');};
  if(editandoId){datos.id=editandoId;google.script.run.withSuccessHandler(onOk).withFailureHandler(onErr).actualizarComanda(datos);}
  else{google.script.run.withSuccessHandler(onOk).withFailureHandler(onErr).crearComanda(datos);}
}

function getAbonoMontoEur(){
  if(abonoMoneda==='EUR')return parseFloat(document.getElementById('abono-monto-eur').value)||0;

  var tasa=parseFloat(document.getElementById('abono-tasa-bs').value)||0;
  return (bs>0&&tasa>0)?bs/tasa:0;
}

/**
 * Renderiza la barra de progreso de envíos para una tarjeta.
 * Retorna HTML string o '' si no hay datos.
 */
function renderProgresoEnvio(cmd){
  var total   = cmd.totalPrendas      || 0;
  var enviado = cmd.prendasEnviadas   || 0;
  var pend    = cmd.prendasPendientes || 0;
  var pct     = cmd.porcentaje        || 0;

  if(total === 0) return '';

  var barClass = pct === 100 ? 'ep-completo' : (pct < 30 ? 'ep-inicio' : '');

  var html = '<div class="envio-progreso">'
    + '<div class="envio-progreso-label">'
    +   '<span>📦 Enviadas</span>'
    +   '<span class="ep-nums">' + enviado + ' / ' + total + '</span>'
    +   (pct > 0 ? '<span class="ep-pct">' + pct + '%</span>' : '')
    + '</div>'
    + '<div class="envio-bar-track">'
    +   '<div class="envio-bar-fill ' + barClass + '" style="width:' + pct + '%"></div>'
    + '</div>';

  if(pend > 0){
    html += '<div class="envio-pendiente-tag">⏳ ' + pend + ' pendiente' + (pend > 1 ? 's' : '') + '</div>';
  }

  html += '</div>';
  return html;
}

/**
 * Renderiza el botón "📦 Enviar parte" para una tarjeta.
 * Solo visible si la comanda no está cancelada/entregada y hay pendientes.
 */
function renderBtnEnviarParte(cmd){
  if(cmd.status === 'cancelado' || cmd.status === 'entregado') return '';
  var hayPend = (cmd.prendasPendientes || 0) > 0 || (cmd.totalPrendas || 0) === 0;
  if(!hayPend) return '';
  return '<button class="btn-enviar-parte" onclick="abrirModalEnvio(\'' + esc(cmd.id) + '\')">'
       +   '📦 Enviar parte'
       + '</button>';
}

/** Renderiza el contenido del body del modal de envío */
function _renderMepBody(cmd){
  var body = document.getElementById('mep-body');
  var productos_cmd = cmd.productos || [];

  // Cuánto ya se envió por SKU
  var yaEnvMap = {};
  (cmd.envios || []).forEach(function(env){
    (env.items || []).forEach(function(item){
      yaEnvMap[item.sku] = (yaEnvMap[item.sku] || 0) + (parseInt(item.cant) || 0);
    });
  });

  // Construir lista de pendientes
  var pendientes = [];
  productos_cmd.forEach(function(p){
    var sku       = p.sku || p.nombre || '';
    var cantTotal = parseInt(p.cant) || parseInt(p.cantidad) || 1;
    var yaEnv     = yaEnvMap[sku] || 0;
    var pend      = Math.max(0, cantTotal - yaEnv);
    if(pend > 0){
      pendientes.push({
        sku:       sku,
        nombre:    p.nombre || p.modelo || sku,
        color:     p.color  || '',
        talla:     p.talla  || '',
        cantTotal: cantTotal,
        yaEnviado: yaEnv,
        pendiente: pend
      });
    }
  });

  var btnConf = document.getElementById('mep-btn-confirmar');

  if(!pendientes.length){
    body.innerHTML = '<div class="mep-empty">✅ Todas las prendas ya fueron enviadas.</div>';
    btnConf.disabled = true;
    return;
  }
  btnConf.disabled = false;

  var hayParciales = (cmd.prendasEnviadas || 0) > 0;
  var html = '';

  if(hayParciales){
    html += '<div class="mep-alerta">'
          + '⚡ Este pedido ya tiene ' + cmd.prendasEnviadas + ' de ' + cmd.totalPrendas
          + ' prendas enviadas. Aquí solo se muestran las pendientes.'
          + '</div>';
  }

  pendientes.forEach(function(item, idx){
    html += '<div class="mep-item" data-sku="' + esc(item.sku) + '">'
          +   '<div class="mep-item-info">'
          +     '<div class="mep-item-nombre">' + esc(item.nombre) + '</div>'
          +     '<div class="mep-item-meta">'
          +       (item.color ? item.color : '')
          +       (item.talla ? ' · T:' + item.talla : '')
          +       ' · SKU: ' + esc(item.sku)
          +     '</div>'
          +   '</div>'
          +   '<div class="mep-item-control">'
          +     '<div class="mep-qty-label">de ' + item.pendiente + '<br>pend.</div>'
          +     '<input type="number" class="mep-qty-input"'
          +       ' id="mep-qty-' + idx + '"'
          +       ' data-sku="' + esc(item.sku) + '"'
          +       ' data-max="' + item.pendiente + '"'
          +       ' min="0" max="' + item.pendiente + '"'
          +       ' value="' + item.pendiente + '"'
          +       ' oninput="validarCantidadEnvio(this)">'
          +   '</div>'
          + '</div>';
  });

  html += '<div class="mep-nota-row">'
        +   '<label class="mep-nota-label">Nota del envío (opcional)</label>'
        +   '<textarea class="mep-nota-input" id="mep-nota" rows="2"'
        +     ' placeholder="Ej: Primer envío, agencia MRW…"></textarea>'
        + '</div>';

  body.innerHTML = html;
}

/** Valida que el input no exceda el máximo permitido */
function validarCantidadEnvio(input){
  var max = parseInt(input.dataset.max) || 0;
  var val = parseInt(input.value)       || 0;
  if(val < 0)   input.value = 0;
  if(val > max) input.value = max;
}

/** Confirma y envía la solicitud al backend */
function confirmarEnvioParcial(){
  if(!_mepComandaActual) return;

  var inputs = document.querySelectorAll('#mep-body .mep-qty-input');
  var items  = [];
  inputs.forEach(function(inp){
    var cant = parseInt(inp.value) || 0;
    if(cant > 0) items.push({ sku: inp.dataset.sku, cant: cant });
  });

  if(!items.length){
    toast('Indica al menos 1 prenda para enviar.', 'err');
    return;
  }

  var nota = (document.getElementById('mep-nota').value || '').trim();
  var btn  = document.getElementById('mep-btn-confirmar');
  btn.disabled    = true;
  btn.textContent = '⏳ Enviando…';

  google.script.run
    .withSuccessHandler(function(res){
      btn.disabled    = false;
      btn.textContent = '✅ Confirmar envío';

      if(!res.ok){
        toast('Error: ' + (res.error || 'Error desconocido'), 'err');
        return;
      }

      cerrarModalEnvio();

      // Actualizar el objeto comanda en memoria para que el re-render sea inmediato
      var cmd = comandas.find(function(c){ return c.id === _mepComandaActual.id; });
      if(cmd){
        cmd.prendasEnviadas   = res.prendasEnviadas;
        cmd.prendasPendientes = res.prendasPendientes;
        cmd.totalPrendas      = res.totalPrendas;
        cmd.porcentaje        = res.porcentaje;
        if(res.autoEntregado){ cmd.status = 'entregado'; cmd.fechaStatus = Date.now(); }
      }
      renderBoard();

      toast(res.mensaje || '📦 Envío registrado.');
      if(res.advertencias && res.advertencias.length){
        res.advertencias.forEach(function(w){ toast('⚠ ' + w, 'err'); });
      }
    })
    .withFailureHandler(function(err){
      btn.disabled    = false;
      btn.textContent = '✅ Confirmar envío';
      toast('Error de conexión: ' + err.message, 'err');
    })
    .registrarEnvioParcial(_mepComandaActual.id, items, nota);
}

/** Cierra el modal de envío parcial */
function cerrarModalEnvio(){
  document.getElementById('modal-envio-parcial').classList.remove('visible');
  _mepComandaActual = null;
}

// Cerrar al hacer clic en el overlay
document.getElementById('modal-envio-parcial').addEventListener('click', function(e){
  if(e.target === this) cerrarModalEnvio();
});
