// ══════════════════════════════════════════
// GENERADOR DE SKU — lógica MODITEX
// =MAYUSC(
//   IZQ(B,2) & EXTRAE(B, LARGO(B)/2, 1) & "-" &
//   IZQ(C,2) & DER(C,1)                 & "-" &
//   IZQ(D,1) & LARGO(E)                 & "-" &
//   TEXTO(FILA*137, "0000")
// )
// ══════════════════════════════════════════
function generarSkuLogica(cat, mod, tal, col, fila) {
  cat  = (cat||'').trim().toUpperCase();
  mod  = (mod||'').trim().toUpperCase();
  tal  = (tal||'').trim().toUpperCase();
  col  = (col||'').trim().toUpperCase();
  fila = parseInt(fila)||2;

  if (!cat && !mod) return null;

  // Parte 1: IZQ(cat,2) + EXTRAE(cat, LARGO(cat)/2, 1)
  var p1 = cat.substring(0,2) + (cat.charAt(Math.floor(cat.length/2))||'');

  // Parte 2: IZQ(mod,2) + DER(mod,1)
  var p2 = mod.substring(0,2) + (mod.charAt(mod.length-1)||'');

  // Parte 3: IZQ(tal,1) + LARGO(col)
  var p3 = (tal.charAt(0)||'U') + col.length;

  // Parte 4: TEXTO(fila*137, "0000")
  var p4 = String((fila*137)%10000).padStart(4,'0');

  return {
    sku: p1+'-'+p2+'-'+p3+'-'+p4,
    explain: [
      'Cat <b>'+p1+'</b> = IZQ("'+cat+'",2)+MEDIO',
      'Mod <b>'+p2+'</b> = IZQ("'+mod+'",2)+DER',
      'T+C <b>'+p3+'</b> = IZQ(talla)+LARGO(color)',
      'Seq <b>'+p4+'</b> = fila('+fila+')×137'
    ].join(' · ')
  };
}

function calcularSku() {
  var r = generarSkuLogica(
    document.getElementById('sku-cat').value,
    document.getElementById('sku-mod').value,
    document.getElementById('sku-tal').value,

    document.getElementById('sku-fila').value
  );
  var box = document.getElementById('sku-result');
  if (!r) { box.style.display='none'; return; }
  box.style.display='flex';
  document.getElementById('sku-result-val').textContent    = r.sku;
  document.getElementById('sku-result-explain').innerHTML  = r.explain;
}

function getBarcodeURL(sku) {
  // paddingheight=0 paddingwidth=0 — elimina espacio blanco interno de la imagen
  // height=12 — barras suficientemente altas para verse bien al escalar
  return 'https://bwipjs-api.metafloor.com/?bcid=code128&text='+
    encodeURIComponent(sku)+'&scale=3&height=12&paddingwidth=0&paddingheight=0';
}

function getSelectedItems() {
  var items=[];
  document.querySelectorAll('.prod-chk:checked').forEach(function(chk){
    var sku=chk.getAttribute('data-sku');
    var qty=parseInt((document.querySelector('.prod-qty[data-sku="'+sku+'"]')||{}).value)||1;
    var p=todosLosProductos.find(function(x){ return x.sku===sku; });
    if (p) for(var i=0;i<qty;i++) items.push(p);
  });
  return items;
}

function generarVista() {
  var items=getSelectedItems();
  if (!items.length){ alert('Selecciona al menos un producto'); return; }
  var layout  = document.getElementById('layout-sel').value;
  var precio  = document.getElementById('precio-sel').value;
  var cfg     = layoutMap[layout];
  var wrap    = document.getElementById('labels-preview-wrap');
  var area    = document.getElementById('preview-area');
  wrap.classList.add('visible');

  var pages=[];
  for(var i=0;i<items.length;i+=cfg.perPage) pages.push(items.slice(i,i+cfg.perPage));

  var html=pages.map(function(pageItems){
    var labelsHTML=pageItems.map(function(p){
      var ph='';
      if(precio==='detal')  ph='<div class="label-precio">'+fmt(p.precioDetal)+'</div>';
      else if(precio==='mayor') ph='<div class="label-precio">'+fmt(p.precioMayor)+' may</div>';
      else if(precio==='ambos') ph='<div class="label-precio">'+fmt(p.precioDetal)+'/'+fmt(p.precioMayor)+'</div>';
      var hex=getColorHex(p.color);
      var nombre=p.modelo+(p.talla&&p.talla!=='UNICA'?' '+p.talla:'');
      // ── NUEVO LAYOUT VERTICAL ──
      // .label-top  → logo (izq) + barcode (todo el ancho restante)
      // .label-sku  → número SKU, debajo, ancho completo
      // .label-nombre → nombre del producto, ancho completo, 1 línea + ellipsis
      // .label-precio → precio (opcional), ancho completo
      return '<div class="label">'+
        '<div class="label-top">'+
          '<div class="label-logo">'+
            '<div class="logo-text">MODITEX</div>'+
            '<div class="logo-dot"></div>'+
            '<div class="logo-grp">GRP</div>'+
          '</div>'+
          '<div class="label-barcode">'+
            '<img src="'+getBarcodeURL(p.sku)+'" alt="'+p.sku+'" loading="lazy">'+
          '</div>'+
        '</div>'+
        '<div class="label-sku">'+p.sku+'</div>'+
        '<div class="label-nombre">'+nombre+'</div>'+
        '<div class="label-color"><span class="label-cdot" style="background:'+hex+'"></span>'+p.color+'</div>'+
        ph+
      '</div>';
    }).join('');
    var empty=cfg.perPage-pageItems.length;
    for(var e=0;e<empty;e++) labelsHTML+='<div class="label" style="border-color:transparent;visibility:hidden"></div>';
    return '<div class="sheet-page layout-'+layout+'">'+labelsHTML+'</div>';
  }).join('');

  area.innerHTML=html;
  document.getElementById('preview-sub').textContent=
    items.length+' etiquetas · '+pages.length+' página'+(pages.length!==1?'s':'')+
    ' — '+layout+' ('+cfg.perPage+'/pág)';
  wrap.scrollIntoView({behavior:'smooth',block:'start'});
}
